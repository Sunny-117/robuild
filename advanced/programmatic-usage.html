程序化使用 ​

除了 CLI，robuild 还提供程序化 API，可以在 Node.js 脚本中调用。

基本用法 ​

scripts/build.ts
ts
import { build } from 'robuild'

await build({
  entries: [
    {
      type: 'bundle',
      input: './src/index.ts',
      format: ['esm', 'cjs'],
      dts: true,
    },
  ],
})

API ​

build(config) ​

执行构建：

scripts/build.ts
ts
import { build } from 'robuild'

await build({
  cwd: '.',
  entries: [
    {
      type: 'bundle',
      input: './src/index.ts',
    },
  ],
  hooks: {
    start: (ctx) => console.log('开始构建:', ctx.pkg.name),
    end: (ctx) => console.log('构建完成'),
  },
})

完整示例 ​

条件构建 ​

scripts/build.ts
ts
import { build } from 'robuild'

const isDev = process.env.NODE_ENV === 'development'

await build({
  entries: [
    {
      type: 'bundle',
      input: './src/index.ts',
      format: ['esm', 'cjs'],
      minify: !isDev,
      sourcemap: isDev,
      dts: !isDev,
    },
  ],
})

多包构建 ​

scripts/build.ts
ts
import { build } from 'robuild'

const packages = ['core', 'utils', 'cli']

for (const pkg of packages) {
  await build({
    cwd: `./packages/${pkg}`,
    entries: [
      {
        type: 'bundle',
        input: './src/index.ts',
        format: ['esm', 'cjs'],
        dts: true,
      },
    ],
  })
}

与其他工具集成 ​

scripts/build.ts
ts
import { build } from 'robuild'
import { execSync } from 'node:child_process'

// 构建
await build({
  entries: [
    {
      type: 'bundle',
      input: './src/index.ts',
    },
  ],
})

// 构建后运行测试
execSync('pnpm test', { stdio: 'inherit' })
